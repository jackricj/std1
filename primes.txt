#!/bin/bash

# Define a function that takes a number as an argument and returns 1 if it is prime, 0 otherwise
is_prime () {
  sqrt=$(awk "BEGIN {printf \"%.2f\n\", sqrt($1)}")
  # Convert the square root to an integer
  sqrtint=${sqrt%.*}
  # Loop over the numbers from 2 to the square root of the number
  for ((i=2; i<=sqrtint; i++))
  do
    # Check if the number is divisible by i
    if (($1%i==0))
    then
      # The number is not prime, return 0
      echo 0
      return
    fi
  done
  # The number is prime, return 1
  echo 1
}

# Ask the user for a number and store it in a variable n
echo "Enter a number:"
read n

# Declare an array to store the prime numbers
declare -a primes

# Loop over the numbers from 2 to n
for ((num=2; num<=n; num++))
do
  # Call the function to check if the number is prime
  result=$(is_prime $num)
  # If the result is 1, add the number to the array
  if ((result==1))
  then
    primes+=($num)
  fi
done

# Print the array of prime numbers
echo "The prime numbers from 2 to $n are:"
echo ${primes[@]}
sleep 10
