Option Explicit

Dim id() As Integer

' Initialize the data structure with N objects
Sub InitializeQuickUnion(N As Integer)
    ReDim id(1 To N)
    Dim i As Integer
    For i = 1 To N
        id(i) = i
    Next i
End Sub

' Find the root of element i
Function Root(i As Integer) As Integer
    Do While i <> id(i)
        id(i) = id(id(i))
        i = id(i)
    Loop
    Root = i
End Function

' Find if p and q are connected
Function Connected(p As Integer, q As Integer) As Boolean
    Connected = (Root(p) = Root(q))
End Function

' Union operation to connect p and q
Sub Union(p As Integer, q As Integer)
    Dim iRoot As Integer
    Dim jRoot As Integer
    iRoot = Root(p)
    jRoot = Root(q)
    id(iRoot) = jRoot
End Sub

' Example usage
Sub ExampleUsage()
    InitializeQuickUnion 10
    Union 1, 2
    Union 3, 4
    Union 4, 5
    Debug.Print Connected(1, 2) ' True
    Debug.Print Connected(1, 3) ' False
    Union 2, 3
    Debug.Print Connected(1, 3) ' True
End Sub
